import tkinter as tk

# Create the main window of an application of a width of half the screen and full height
root = tk.Tk()
root.title("Feast")
root.geometry("960x1080")

# Create a frame for the main window
frame = tk.Frame(root)
frame.pack()

# Create a label for the title of the application
title = tk.Label(frame, text="Feast", font=("Helvetica", 32))
title.pack()

# Create a label for the description of the application
description = tk.Label(frame, text="A simple recipe application", font=("Helvetica", 16))
description.pack()

# Create a button for the user to add a recipe
add_recipe = tk.Button(frame, text="Add Recipe")
add_recipe.pack()

# Create a button for the user to view a recipe
view_recipe = tk.Button(frame, text="View Recipe")
view_recipe.pack()

# Create a button for the user to edit a recipe
edit_recipe = tk.Button(frame, text="Edit Recipe")
edit_recipe.pack()

# Create a button for the user to delete a recipe
delete_recipe = tk.Button(frame, text="Delete Recipe")
delete_recipe.pack()

# Create a button for the user to exit the application
exit = tk.Button(frame, text="Exit")
exit.pack()

# Center the application on the screen by setting the x and y coordinates to half the screen size

# Set the x and y coordinates of the application to half the screen size at the top height of the screen
x_coordinate = int(root.winfo_screenwidth() / 4)
y_coordinate = int(root.winfo_screenheight())
 
# Set the x and y coordinates of the application to half the screen size at the top height of the screen
root.geometry("+{}+{}".format(x_coordinate, y_coordinate))


# Run the application
root.mainloop()
